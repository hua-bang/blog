еҶ’жіЎжҺ’еәҸ вҖӢ

еҶ’жіЎжҺ’еәҸпјҲbubble sortпјүйҖҡиҝҮиҝһз»ӯең°жҜ”иҫғдёҺдәӨжҚўзӣёйӮ»е…ғзҙ е®һзҺ°жҺ’еәҸгҖӮе®һйҷ…дёҠпјҢдјҡејҖеҗҜдёҖж¬ЎеҫӘзҺҜпјҢжҜҸдёҖж¬ЎеҫӘзҺҜдёӯпјҢдјҡд»Һе·ҰеҲ°еҸіпјҢе°ҶдёӨдёӘж•°з»„е…ғзҙ жҜ”иҫғпјҢе°ҶжҜ”иҫғеӨ§зҡ„е…ғзҙ з§»еҲ°еҸіиҫ№еҺ»гҖӮжҜҸж¬ЎеҫӘзҺҜеҗҺпјҢеӣһеҮҸе°ҸдёҖдёӘеҢәй—ҙиҢғеӣҙгҖӮ

жү§иЎҢжөҒзЁӢ вҖӢ

еҰӮж•°з»„ [4,5,3,1], зҡ„дёҖж¬ЎеҫӘзҺҜжү§иЎҢеҰӮдёӢгҖӮ иҝҷж ·еӯҗпјҢжҲ‘д»¬еҸҜд»ҘзңӢеҲ°пјҢжҜҸж¬ЎеҫӘзҺҜжңҖеӨ§зҡ„е…ғзҙ дёҖе®ҡдјҡеҲ°жңҖеҸіиҫ№гҖӮ

ж•ҙдҪ“жөҒзЁӢ

д»Јз Ғе®һзҺ° вҖӢ

ts
const bubbleSort = (arr: number[]) => {
  for (let i = arr.length - 1; i > 0; i--) {
    for (let j = 0; j < i; j++) {
      if (arr[j] > arr[j + 1]) {
        [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
      }
    }
  }
};

ж•ҲзҺҮдјҳеҢ– вҖӢ

жҲ‘д»¬еҸҜд»ҘеңЁеҫӘзҺҜдёӯпјҢеҠ дёҖдёӘ flag, иЎЁзӨәеҪ“еүҚдёҖж¬ЎеҫӘзҺҜжңүжІЎжңүжӣҝжҚўиҝҮйЎәеәҸпјҢжІЎжңүзҡ„иҜқпјҢеҲҷиҜҙжҳҺпјҢе·Із»Ҹе®ҢжҲҗдәҶгҖӮ

ts
const bubbleSort = (arr: number[]) => {
  for (let i = arr.length - 1; i > 0; i--) {
    let flag = false;
    for (let j = 0; j < i; j++) {
      if (arr[j] > arr[j + 1]) {
        [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
        flag = true;
      }
    }
    if (!flag) {
      break;
    }
  }
};
